    img.onmousedown = function(e) {

        document.body.appendChild(img);
        img.style.zIndex = 1000;

        function moving(e) {
            img.style.left = e.pageX - img.offsetWidth / 2 + 'px';
            img.style.top = e.pageY - img.offsetHeight / 2 + 'px';
        }
        document.onmousemove = function(e) {
